291 // Fill holes
292 template<typename... Args>
293 Computation& fill(const std::string& name, double value, Args... args) {
294 std::unordered_map<std::string, secure::ValuePtr> values;
295 values[name] = secure::num(value);
296
297 // Recursively fill remaining holes
298 if constexpr (sizeof...(args) > 0) {
299 fill_impl(values, args...);

303 if (!filled) {

309 }

312 Computation& fill(const std::unordered_map<std::string, double>& values) {
313 std::unordered_map<std::string, secure::ValuePtr> val_map;
314 for (const auto& [name, val] : values) {
315 val_map[name] = secure::num(val);
316 }

319 if (!filled) {

321 }

325 }

341 values[name] = secure::num(value);

377 // Verify expression hasn't been tampered with
378 bool verify(const Expr& received) {
379 auto holes_original = expression.get_holes();
380 auto holes_received = received.get_holes();
381
382 // Check all original holes are preserved
383 for (const auto& hole : holes_original) {
384 if (std::find(holes_received.begin(), holes_received.end(), hole)
